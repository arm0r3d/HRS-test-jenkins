pipeline {
    agent any
    stages {
        stage('Restore') {
            steps {
                bat 'dotnet restore HouseRentingSystem.sln'
            }
        }
        stage('Build'){
            steps {
                bat build 'dotnet build HouseRentingSystem.sln --configuration Release'
            }
        }
        stage('Test'){
            steps {
                bat build 'dotnet test HouseRentingSystem.Test/HouseRentingSystem.Test.csproj'
            }
        }
    }
}
